#!/bin/bash

# Quick deployment commands for BarangayCare Android app

echo "========================================="
echo "BarangayCare Android - Quick Deploy Guide"
echo "========================================="
echo ""

# Check if ADB is available

if ! command -v adb &> /dev/null; then
echo "⚠️ ADB not found in PATH"
echo "Please install Android SDK Platform-Tools:"
echo " https://developer.android.com/tools/releases/platform-tools"
exit 1
fi

echo "📱 Android Device Management"
echo "================================"
echo ""
echo "1. List Connected Devices:"
echo " adb devices"
echo ""
echo "2. Install App:"
echo " adb install -r android/app/build/outputs/apk/debug/app-debug.apk"
echo ""
echo "3. Uninstall App:"
echo " adb uninstall com.barangaycare.app"
echo ""
echo "4. View App Logs:"
echo " adb logcat | grep -i barangaycare"
echo ""
echo "5. Launch App:"
echo " adb shell am start -n com.barangaycare.app/.MainActivity"
echo ""
echo "6. Clear App Data:"
echo " adb shell pm clear com.barangaycare.app"
echo ""

echo "🔨 Build Commands"
echo "================================"
echo ""
echo "1. Clean Build:"
echo " cd android && ./gradlew.bat clean"
echo ""
echo "2. Build Debug APK:"
echo " cd android && ./gradlew.bat assembleDebug"
echo ""
echo "3. Build & Install (one command):"
echo " cd android && ./gradlew.bat installDebug"
echo ""
echo "4. Rebuild Web Assets:"
echo " npm run build && npx cap sync android"
echo ""

echo "📦 APK Information"
echo "================================"
echo ""
echo "APK Path: android/app/build/outputs/apk/debug/app-debug.apk"
echo "Size: 4.5 MB"
echo "App ID: com.barangaycare.app"
echo "Version: 1.0 (Debug)"
echo ""

echo "🔔 Notification Testing"
echo "================================"
echo ""
echo "Admin Test:"
echo " 1. Log in as admin"
echo " 2. Have another user submit complaint"
echo " 3. Should receive: 'New complaint submitted' notification"
echo ""
echo "Resident Test:"
echo " 1. Log in as resident"
echo " 2. Submit a complaint"
echo " 3. Should receive: 'Complaint received' notification"
echo " 4. When admin updates status → 'Complaint status updated'"
echo ""

echo "📋 Troubleshooting"
echo "================================"
echo ""
echo "Device Not Detected:"
echo " - Disable and re-enable USB Debugging"
echo " - Disconnect and reconnect USB cable"
echo " - Check drivers: adb devices"
echo ""
echo "App Crashes:"
echo " - View logs: adb logcat -s barangaycare"
echo " - Check Supabase connection"
echo " - Verify device has internet"
echo ""
echo "Notifications Not Working:"
echo " - Check permissions: Settings → Apps → BarangayCare → Permissions"
echo " - Ensure 'Notifications' permission is granted"
echo " - Restart app after granting permissions"
echo ""

echo "✅ Ready to Deploy!"
echo ""
echo "Next step: Connect your Android device and run:"
echo " adb install -r android/app/build/outputs/apk/debug/app-debug.apk"
echo ""
